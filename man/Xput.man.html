NAME

XputChunk, XputFile, XputBuffer − Content Publishing Functions.

SYNOPSIS

#include <xcache.h>

int XputChunk(int xcacheSock, ChunkContext *ctx, const void *data,
size_t length, ChunkInfo *info)
int XputFile(int xcacheSock, ChunkContext *ctx, const char *filename,
size_t chunkSize, ChunkInfo **info)
int XputBuffer(int xcacheSock, ChunkContext *ctx, const void *data,
size_t length, size_t chunkSize, ChunkInfo **info)

DESCRIPTION

The functions in Xput family publish content to the XIA network. The content chunks are added into the slice pointed to by ctx argument. (See Xslice(8) for details).

XputChunk() publishes a single chunk of data pointed to by argument data and having length length. It fills info with following information about the chunk just published.

typedef struct {

int size;

char cid[CID_HASH_SIZE + 1];

int32_t ttl; /* FIXME: unused */

struct timeval timestamp; /* FIXME: unused */

} ChunkInfo;

XputFile() publishes content from file filename while XputBuffer publishes content from buffer data of total length length. Both the functions publish chunks of size chunkSize at the max into the cache slice ctx which should have been allocated by XallocSlice(8). Information about the chunks published is filled in info which is allocated by XputBuffer() or XputFile().

RETURN VALUE

On success, all these functions return number of chunks published. Return value 0 does not indicate any error. On failure, these functions return -1 and errno is set appropriately.

ERRORS

ENOTCONN

XcacheSock is not connected.

EHOSTDOWN

Xcache is not running.

EXAMPLE

#include <xcache.h>
#include <stdio.h>

int main(void)
{

int sock = XcacheSock();

ChunkCtx *ctx;

struct sliceInfo info;

char buffer[1024];

ChunkInfo *info;

int n_chunks;

if(sock < 0) {

perror("XcacheSock");

exit(EXIT_FAILURE);

}

info.ttl = XCACHE_INF;

info.size = XCACHE_INF;

info.policy = XCACHE_EVICT_POLICY_LRU;

ctx = XallocateSlice(sock, &info);

if(!ctx) {

perror("XallocateSlice");

exit(EXIT_FAILURE);

}

memset(buffer, 0, sizeof(buffer));

if((n_chunks = XputBuffer(sock, ctx, buffer, 1024, 100, &info)) < 0) {

perror(XputBuffer());

} else {

printf("Published %d chunks successfully!\n", n_chunks);

}

if(XfreeSlice(sock, ctx) < 0) {

perror("XfreeSlice");

}

close(sock);

}
